const chai = require("chai");
const expect = chai.expect;
const isTriangle = require("../src/triangleDeterminer");
 
describe("Triangle Determiner", function () {
  // Тесты для случаев, когда треугольник существует
  describe("Valid Triangles", function () {
    it("should return true for valid triangle with sides 3, 4, 5", function () {
      expect(isTriangle(3, 4, 5)).to.be.true;
    });
 
    it("should return true for valid triangle with sides 5, 12, 13", function () {
      expect(isTriangle(5, 12, 13)).to.be.true;
    });
 
    it("should return true for valid triangle with sides 8, 15, 17", function () {
      expect(isTriangle(8, 15, 17)).to.be.true;
    });
  });
 
  // Тесты для случаев, когда треугольник не существует
  describe("Invalid Triangles", function () {
    it("should return false for sides 1, 1, 3", function () {
      expect(isTriangle(1, 1, 3)).to.be.false;
    });
 
    it("should return false for sides 0, 0, 0", function () {
      expect(isTriangle(0, 0, 0)).to.be.false;
    });
 
    it("should return false for sides 2, 4, 6", function () {
      expect(isTriangle(2, 4, 6)).to.be.false;
    });
  });
 
  // Тесты для граничных случаев
  describe("Edge Cases", function () {
    it("should return false for sides 1, 2, Infinity", function () {
      expect(isTriangle(1, 2, Infinity)).to.be.false;
    });
 
    it("should return false for sides 1, 2, NaN", function () {
      expect(isTriangle(1, 2, NaN)).to.be.false;
    });
 
    it("should return false for sides -1, 2, 3", function () {
      expect(isTriangle(-1, 2, 3)).to.be.false;
    });
  });
});
